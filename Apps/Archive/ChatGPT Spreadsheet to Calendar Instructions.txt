MONTHS = {"JAN":1,"FEB":2,"MAR":3,"APR":4,"MAY":5,"JUN":6,"JUL":7,"AUG":8,"SEP":9,"OCT":10,"NOV":11,"DEC":12}

def cell_text(cell):
    ps = cell.getElementsByType(P)
    out = []
    for p in ps:
        for n in p.childNodes:
            if getattr(n, "data", None):
                out.append(n.data)
    return re.sub(r"\s+", " ", "".join(out)).strip()

def expand_row(row):
    out = []
    for cell in row.getElementsByType(TableCell):
        rep = int(cell.getAttribute("numbercolumnsrepeated") or 1)
        out.extend([cell] * rep)
    return out

def expand_rows(table):
    rows = []
    for row in table.getElementsByType(TableRow):
        rep = int(row.getAttribute("numberrowsrepeated") or 1)
        expanded = expand_row(row)
        for _ in range(rep):
            rows.append(expanded)
    return rows

def canon_name(s):
    if not s: return None
    t = re.sub(r"\s+", " ", s.strip()).upper()
    t = re.sub(r"[^\w\s].*$", "", t).strip()
    return t if t in ALLOWED else None

def filename_name(canon):
    parts = canon.split()
    return "".join(p[:1] + p[1:].lower() for p in parts)

def parse_month_year(title):
    m = re.search(r"\b(JAN|FEB|MAR|APR|MAY|JUN|JUL|AUG|SEP|OCT|NOV|DEC)\b[^0-9]*(\d{4})", title, re.I)
    if not m: return None, None
    return m.group(1).upper(), int(m.group(2))

def parse_mmdd_from_header(h, year):
    m = re.search(r"\b(\d{1,2})/(\d{1,2})\b", h)
    if not m: return None
    mm, dd = int(m.group(1)), int(m.group(2))
    return date(year, mm, dd)

TIME_TOKEN_RE = re.compile(r"(?:\d{1,2}:\d{2}[AP]|\d{1,4}[AP])", re.I)

def parse_time_token(tok):
    if not tok: return None
    t = tok.strip().upper().replace(" ", "")
    if t == "CLOSE":
        return (14, 0)
    t = re.sub(r"^(OPEN|CLOSE):", "", t)
    m = re.match(r"^(\d{1,4})(?::?(\d{2}))?(A|P)$", t)
    if not m: return None
    digits, mm_str, ap = m.group(1), m.group(2), m.group(3)
    if mm_str is not None:
        h, mi = int(digits), int(mm_str)
    else:
        if len(digits) in (3, 4):
            h, mi = int(digits[:-2]), int(digits[-2:])
        else:
            h, mi = int(digits), 0
    if mi > 59 or h > 12 or h < 0:
        return None
    if ap == "A":
        if h == 12: h = 0
    else:
        if h != 12: h += 12
    return (h, mi)

DEFAULT_END = {
    (6,0):(14,30),
    (7,0):(15,30),
    (8,0):(16,30),
    (8,30):(17,0),
    (9,0):(17,30),
    (14,0):(22,30),
}

def parse_shift_text(shift_text):
    if not shift_text: return None
    s = re.sub(r"\s+", " ", shift_text.strip())
    if re.match(r"^(OFF|PTO|HOLIDAY)\b", s, re.I):
        return {"kind":"allday", "summary": s}
    if re.search(r"\b(OFF|PTO|HOLIDAY)\b", s, re.I) and not re.search(r"\d", s):
        return {"kind":"allday", "summary": s}
    m = re.search(r"(" + TIME_TOKEN_RE.pattern + r"|CLOSE)\s*-\s*(" + TIME_TOKEN_RE.pattern + r")", s, re.I)
    if m:
        st, en = parse_time_token(m.group(1)), parse_time_token(m.group(2))
        if st and en:
            return {"kind":"timed", "start": st, "end": en, "raw": s}
    m = re.search(r"\b(OPEN:\s*" + TIME_TOKEN_RE.pattern + r"|CLOSE:\s*" + TIME_TOKEN_RE.pattern + r"|CLOSE|" + TIME_TOKEN_RE.pattern + r")\b", s, re.I)
    if m:
        st = parse_time_token(m.group(1))
        if st:
            en = DEFAULT_END.get(st)
            if en is None:
                raise RuntimeError(f"TOOLS FAILED: missing default end mapping for start {st} in cell '{shift_text}'")
            return {"kind":"timed", "start": st, "end": en, "raw": s}
    return None

def has_oncall(s):
    return bool(s and re.search(r"\bON\s*CALL\b|\bONCALL\b", s, re.I))

def monday_of(d):
    return d - timedelta(days=d.weekday())

def ics_escape(text):
    if text is None: return ""
    text = text.replace("\\", "\\\\")
    text = text.replace(";", "\\;").replace(",", "\\,")
    text = text.replace("\r\n", "\n").replace("\r", "\n")
    text = text.replace("\n", "\\n")
    return text

def fold_ical_line(line, limit=75):
    if len(line) <= limit: return [line]
    out = [line[:limit]]
    rest = line[limit:]
    while rest:
        out.append(" " + rest[:limit-1])
        rest = rest[limit-1:]
    return out

def dtstamp_now():
    return datetime.utcnow().strftime("%Y%m%dT%H%M%SZ")

def fmt_date(d): return d.strftime("%Y%m%d")
def fmt_dt_local(d, hm):
    return datetime(d.year, d.month, d.day, hm[0], hm[1], 0).strftime("%Y%m%dT%H%M%S")

def build_ics(events):
    lines = [
        "BEGIN:VCALENDAR","VERSION:2.0","CALSCALE:GREGORIAN","METHOD:PUBLISH",
        "PRODID:-//Sterling Specialty//RPH Schedule Export//EN"
    ]
    for ev in events:
        lines.append("BEGIN:VEVENT")
        lines.append(f"UID:{ev['uid']}")
        lines.append(f"DTSTAMP:{ev['dtstamp']}")
        if ev["allday"]:
            lines.append(f"DTSTART;VALUE=DATE:{ev['dtstart']}")
            lines.append(f"DTEND;VALUE=DATE:{ev['dtend']}")
        else:
            lines.append(f"DTSTART;TZID={TZID}:{ev['dtstart']}")
            lines.append(f"DTEND;TZID={TZID}:{ev['dtend']}")
        lines.append(f"SUMMARY:{ics_escape(ev['summary'])}")
        desc = ev.get("description") or ""
        if desc:
            lines.append(f"DESCRIPTION:{ics_escape(desc)}")
        lines.append("END:VEVENT")
    lines.append("END:VCALENDAR")
    folded = []
    for ln in lines:
        folded.extend(fold_ical_line(ln))
    return "\r\n".join(folded) + "\r\n"

def find_event(nm, d, timed=None):
    for ev in events_by[nm]:
        if ev["_date"] == d and (timed is None or (not ev["allday"]) == timed):
